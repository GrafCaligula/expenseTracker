UI/Modals/Backdrop.js

import React from 'react';

import { useDispatch } from 'react-redux';
import { expActions } from '../../../store/expenses-slice';

import classes from './Backdrop.module.css';

/**
 * A backdrop to block user inputs when a modal is shown
 * @Component 
 */
const Backdrop = () => {
  // Hooks
  /**
   * import to dispatch action against the store
   */
  const dispatch = useDispatch();

  /**
   * function to close the shown modal and backdrop
   */
  const closeBackdropHandler = () => {
    dispatch(expActions.toggleBackdrop());
    dispatch(expActions.clearError());
  };

  return <div className={classes.backdrop} onClick={closeBackdropHandler} />;
};

export default Backdrop;
